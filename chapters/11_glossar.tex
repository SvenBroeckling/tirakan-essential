\worldchapter{Glossar}{license}
\label{ch:glossar}

\begin{multicols*}{2}
\section{Begriffe}\label{sec:begriffe}
\begin{description}[style=nextline,leftmargin=0pt]
\item[\textbf{Arcana}] Die Wissensfertigkeit fur okkulte Analyse; nicht zu verwechseln mit dem Arcana Pool.
\item[\textbf{Arkana}] Die Ressource, mit der Zauberei bezahlt wird.
\item[\textbf{Schutz}] Rüstungswert, der eingehenden Schaden einmalig reduziert.
\item[\textbf{Assist}] Aktion, die +10 auf einen unmittelbaren Test eines Verbündeten gibt.
\item[\textbf{Bindung}] Eine lebendige Bindung an Person, Ort, Orden oder Pflicht.
\item[\textbf{Breaking Point}] Zustand, wenn Burden exakt das Limit erreicht.
\item[\textbf{Burden}] Druckleiste fur Stress, Erschöpfung, Schuld und Korruptionslast.
\item[\textbf{Weg}] Berufung oder sozialer Hintergrund eines Charakters.
\item[\textbf{Catastrophic}] Die höchste Eskalationsstufe eines riskant gewählten Zaubers und zugleich die schlimmste Misserfolgsbandbreite bei 96--00.
\item[\textbf{Consequence}] Preis einer Handlung auf den Stufen Minor, Serious oder Critical.
\item[\textbf{Corruption}] Langzeitdruck durch verbotene übernatürliche Praxis.
\item[\textbf{Dying}] Zustand oberhalb des Wound Threshold; ohne Hilfe tödlich.
\item[\textbf{Engaged, Near, Far, Distant}] Die vier relativen Distanzsegmente im Konflikt.
\item[\textbf{Glaubensniveau}] Epochenwert, der Omen-Maximum und Anrufungen mitbestimmt.
\item[\textbf{Faith Rank}] Rang der Fertigkeit Faith.
\item[\textbf{Fast Cast}] Nicht-rituelles Zaubern mit Kosten von 1 Action.
\item[\textbf{Full Rest}] Sichere volle Ruhephase mit besserer Heilung und voller Poolregeneration.
\item[\textbf{Gear Broken}] Zustand, in dem ein Ausrustungsstuck unbrauchbar ist.
\item[\textbf{Gunst}] Ressource für Anrufungen und Glaubenswunder.
\item[\textbf{Great}] Hohe Wirkstufe eines Erfolgs.
\item[\textbf{Guide}] Kanonischer Begriff für die Spielleitung.
\item[\textbf{Growth}] Fortschrittsmarke für überstandene Kosten und Verwundbarkeit.
\item[\textbf{Schaden}] Grundschaden einer Waffe oder eines Effekts.
\item[\textbf{Initiative Target}] Zielwert für die Reihenfolge zu Kampfbeginn.
\item[\textbf{Abstammung}] Abstammungs-Prägung eines Charakters.
\item[\textbf{Magieniveau}] Epochenwert, der die Zauberverstärkung bestimmt.
\item[\textbf{Prägung}] Sammelbegriff für Abstammung, Weg, Bindung und Mal.
\item[\textbf{Mark Edge Sentence}] Schmaler einmal-pro-Sitzung-Vorteil, der an ein Mark gekoppelt ist.
\item[\textbf{Omen}] Seltene Meta-Ressource fur Eingriffe in Schicksal und Timing.
\item[\textbf{Petition}] Regelverfahren fur Glaubensbitten.
\item[\textbf{Quick Reaction}] Vorab-Reaktion vor der ersten Kampfrunde.
\item[\textbf{Reaction}] Abwehr- oder Antwortaktion, die aus gehaltenen Actions bezahlt wird.
\item[\textbf{Mal}] Verdiente Narbe mit dauerhaftem Vorteil und Verwundbarkeit.
\item[\textbf{Seal}] Rustungseigenschaft gegen Shaken und Bleeding.
\item[\textbf{Short Rest}] Kurze sichere Pause mit begrenzter Heilung.
\item[\textbf{Sign, Intercession, Theophany}] Die positiven Ergebnisbander einer Petition.
\item[\textbf{Situation}] Die Risikoeinordnung eines Tests: Controlled, Risky oder Desperate.
\item[\textbf{Zauberei}] Regelverfahren fur direkte Zauberei.
\item[\textbf{Zauberverstärkung}] Verstärkung gelungener Zauber, wird vom Magieniveau beeinflusst.
\item[\textbf{Strained Bond}] Vorubergehend erschopfte Bindung ohne Bonus.
\item[\textbf{Test}] Standardwurf des Spiels mit 1d100.
\item[\textbf{Oberhand}] Kleiner Vorteil bei beidseitigem Scheitern einer Gegenprobe.
\item[\textbf{Wille}] Attribut für Disziplin, Ausdauer und Überzeugung.
\item[\textbf{Wundenschwelle}] Grenze, ab der weitere Wunden einen Charakter in Sterbend bringen.
\item[\textbf{Wunden}] Druckleiste für körperlichen Schaden.
\end{description}
\end{multicols*}
