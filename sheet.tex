\newcommand{\panelheader}[1]{%
    \noindent\colorbox{ruleaccent}{\parbox{\dimexpr\linewidth-2\fboxsep\relax}{\color{white}\displayfont\large #1}}\par
}

\newcommand{\fieldwithheading}[2][1.2cm]{%
    \parbox[t]{\linewidth}{%
        \textbf{#2}\par
        \vspace{0.22cm}
        \rule{\linewidth}{0.4pt}\vspace{#1}%
    }%
}

\newcommand{\fieldwithcheckboxes}[3][0pt]{%
    \parbox[t]{\linewidth}{%
        \textbf{#2}\hfill\trackboxes{#3}\vspace{#1}%
    }%
}

\newcommand{\tinybox}[1]{%
    \tikz[baseline=(b.base)]\node[
        draw=boxline,
        fill=white,
        minimum width=1.0em,
        minimum height=1.0em,
        inner sep=0pt
    ] (b) {\strut #1};
}

\newcommand{\trackboxes}[1]{%
    \foreach \n in {0,...,\numexpr#1-1\relax}{%
        \tinybox{\textcolor{boxline!55}{\fontsize{5.5pt}{5.5pt}\selectfont \n}}%
    }%
}
